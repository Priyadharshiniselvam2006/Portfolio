Mandelbrot Set Generator

Complex number iteration algorithm

def mandelbrot(c, max_iter=100):
    """
    Determine if complex number c is in Mandelbrot set
    z(n+1) = z(n)² + c
    """
    z = 0
    for n in range(max_iter):
        if abs(z) > 2:
            return n
        z = z*z + c
    return max_iter

def generate_fractal(width, height, zoom=1, center=0):
    """Generate Mandelbrot fractal image"""
    fractal = np.zeros((height, width))
    
    for y in range(height):
        for x in range(width):
            # Map pixel to complex plane
            real = (x - width/2) / (width/4) / zoom + center.real
            imag = (y - height/2) / (height/4) / zoom + center.imag
            c = complex(real, imag)
            
            fractal[y, x] = mandelbrot(c)
    
    return fractal

Statistical Analysis

Hypothesis testing implementation

class StatisticalAnalysis {
    constructor(data) {
        this.data = data;
        this.n = data.length;
    }
    
    mean() {
        return this.data.reduce((sum, x) => sum + x, 0) / this.n;
    }
    
    variance() {
        const mu = this.mean();
        return this.data.reduce((sum, x) => sum + (x - mu) ** 2, 0) / this.n;
    }
    
    tTest(hypothesizedMean, alpha = 0.05) {
        const sampleMean = this.mean();
        const sampleVar = this.variance();
        const standardError = Math.sqrt(sampleVar / this.n);
        
        // Calculate t-statistic
        const t = (sampleMean - hypothesizedMean) / standardError;
        
        // Degrees of freedom
        const df = this.n - 1;
        
        return {
            tStatistic: t,
            degreesOfFreedom: df,
            pValue: this.calculatePValue(t, df),
            significant: Math.abs(t) > this.criticalValue(alpha, df)
        };
    }
}
