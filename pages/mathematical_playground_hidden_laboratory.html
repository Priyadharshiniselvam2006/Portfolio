Mandelbrot Set Generator

function mandelbrot(c, maxIter) {
  let z = { real: 0, imag: 0 };
  let iter = 0;
  
  while (iter < maxIter) {
    const zReal = z.real * z.real - z.imag * z.imag + c.real;
    const zImag = 2 * z.real * z.imag + c.imag;
    
    if (zReal * zReal + zImag * zImag > 4) break;
    
    z.real = zReal;
    z.imag = zImag;
    iter++;
  }
  
  return iter;
}

Prime Number Sieve

function sieveOfEratosthenes(limit) {
  const primes = new Array(limit + 1).fill(true);
  primes[0] = primes[1] = false;
  
  for (let i = 2; i * i <= limit; i++) {
    if (primes[i]) {
      for (let j = i * i; j <= limit; j += i) {
        primes[j] = false;
      }
    }
  }
  
  return primes.map((isPrime, num) => 
    isPrime ? num : null
  ).filter(Boolean);
}
